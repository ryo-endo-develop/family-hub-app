```sql
-- 家族マスター
CREATE TABLE families (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    family_name VARCHAR(100) NOT NULL,             -- 家族名
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- ユーザーマスター (OIDC連携)
CREATE TABLE users (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    oidc_subject VARCHAR(255) NOT NULL UNIQUE,   -- OIDCプロバイダーからの一意な識別子
    email VARCHAR(255) UNIQUE NULL,             -- OIDCから取得できれば
    name VARCHAR(100) NULL,                    -- OIDCから取得できれば
    avatar_url VARCHAR(255) NULL,               -- OIDCから取得できれば
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 家族への所属情報 (中間テーブル)
CREATE TABLE family_memberships (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,           -- usersテーブルのID
    family_id BIGINT UNSIGNED NOT NULL,         -- familiesテーブルのID
    role ENUM('admin', 'member') NOT NULL DEFAULT 'member', -- 家族内での役割
    joined_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    UNIQUE KEY uk_user_family (user_id, family_id), -- 同一ユーザーは同一家族に1回だけ所属
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (family_id) REFERENCES families(id) ON DELETE CASCADE
);

-- ラベルマスター (家族ごとに管理)
CREATE TABLE labels (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    family_id BIGINT UNSIGNED NOT NULL,             -- どの家族のラベルか
    name VARCHAR(50) NOT NULL,                     -- ラベル名
    color VARCHAR(7) NULL,                         -- 選択式カラーパレットの色コード (例: '#FFB3BA')
    created_by_id BIGINT UNSIGNED NULL,            -- 作成したユーザーのID
    updated_by_id BIGINT UNSIGNED NULL,            -- 最後に更新したユーザーのID
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uk_family_label_name (family_id, name), -- 同一家族内では同じラベル名は不可
    FOREIGN KEY (family_id) REFERENCES families(id) ON DELETE CASCADE,
    FOREIGN KEY (created_by_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (updated_by_id) REFERENCES users(id) ON DELETE SET NULL
);

-- タスク本体 (家族ごとに管理)
CREATE TABLE tasks (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    family_id BIGINT UNSIGNED NOT NULL,             -- どの家族のタスクか
    title VARCHAR(255) NOT NULL,
    is_done BOOLEAN NOT NULL DEFAULT FALSE,
    task_type ENUM('single', 'routine') NOT NULL,
    due_date DATE NULL,
    next_occurrence_date DATE NULL,                -- 定常タスクの次回実行日
    routine_settings JSON NULL,                    -- 定常タスクの繰り返し設定 (JSON)
    assignee_id BIGINT UNSIGNED NULL,              -- 担当者のユーザーID
    notes TEXT NULL,                               -- メモ
    priority TINYINT UNSIGNED NULL,                -- 優先度 (例: 1:高, 2:中, 3:低)
    parent_task_id BIGINT UNSIGNED NULL,           -- 親タスクのID (サブタスク用)
    created_by_id BIGINT UNSIGNED NULL,            -- 作成者のユーザーID
    updated_by_id BIGINT UNSIGNED NULL,            -- 最終更新者のユーザーID
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (family_id) REFERENCES families(id) ON DELETE CASCADE,
    FOREIGN KEY (assignee_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (parent_task_id) REFERENCES tasks(id) ON DELETE CASCADE,
    FOREIGN KEY (created_by_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (updated_by_id) REFERENCES users(id) ON DELETE SET NULL
);

-- タスクとラベルの関連 (中間テーブル)
CREATE TABLE task_labels (
    task_id BIGINT UNSIGNED NOT NULL,      -- tasksテーブルのID
    label_id BIGINT UNSIGNED NOT NULL,     -- labelsテーブルのID
    PRIMARY KEY (task_id, label_id),       -- 複合主キー
    FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE,
    FOREIGN KEY (label_id) REFERENCES labels(id) ON DELETE CASCADE
);
```
